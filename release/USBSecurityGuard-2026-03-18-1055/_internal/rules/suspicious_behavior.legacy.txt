rule Suspicious_Script_Commands
{
    meta:
        description = "Detects suspicious script execution"
        author = "USB Security Guard"
        date = "2024"
    strings:
        $script1 = "powershell" nocase ascii wide
        $script2 = "cmd /c" nocase ascii wide
        $script3 = "bash -i" nocase ascii wide
        $script4 = "python -m" nocase ascii wide
    condition:
        any of them
}
